## 集合列表

本项目使用 6 个集合：

| 集合名称 | 用途 | 记录数（预计） | 权限 |
|----------|------|---------------|------|
| `users` | 用户信息 | 用户数 | 创建者可写，所有可读 |
| `checkins` | 打卡记录 | 5000+ | 创建者可写，所有可读 |
| `museums` | 博物馆数据 | 1000+ | 创建者可写，所有可读 |
| `themeParks` | 游乐场数据 | 18 | 创建者可写，所有可读 |
| `userFavorites` | 用户收藏/想去 | 用户数 × 场所数 | 创建者可写，所有可读 |
| `dataSyncLog` | 数据同步日志 | 100+ | 创建者可写，所有可读 |

## 数据迁移计划

### 初始化数据

1. **游乐场数据**
   - 来源：`data/theme-parks.json`
   - 数量：18 条
   - 初始化云函数：`syncThemeParkData`
   - 执行时机：项目部署后立即执行

2. **博物馆数据**
   - 来源：GitHub API
   - 数量：1000+ 条
   - 初始化云函数：`syncMuseumData`
   - 执行时机：项目部署后立即执行

### 数据同步策略

- **同步频率：** 每周一次（7 天）
- **同步时间：** 凌晨 3:00
- **错误重试：** 自动重试 3 次，每次间隔 1 小时
- **同步日志：** 每次同步都记录到 `dataSyncLog` 表

---

## 数据备份计划

### 备份策略

- **自动备份：** 每日备份一次
- **保留周期：** 保留最近 30 天的备份
- **备份存储：** 云存储中的 `backup/` 目录

### 恢复策略

- **恢复测试：** 每月测试一次恢复流程
- **数据验证：** 恢复后验证数据完整性
- **故障切换：** 如主数据库故障，10 分钟内切换到备份

---

## 性能优化建议

### 查询优化

1. **使用索引查询**
   - 尽量使用已创建的索引字段进行查询
   - 避免全表扫描

2. **分页查询**
   - 使用 `limit()` 和 `skip()` 进行分页
   - 每页不超过 50 条记录

3. **选择性查询**
   - 只查询需要的字段
   - 使用 `field()` 限制返回字段

### 写入优化

1. **批量写入**
   - 使用 `add()` 批量插入数据
   - 单次批量不超过 500 条

2. **异步写入**
   - 使用异步操作提高并发
   - 避免阻塞主线程

## 附录A：数据库初始化脚本

### 创建集合和索引

```javascript
const cloud = require('wx-server-sdk');
cloud.init({ env: cloud.DYNAMIC_CURRENT_ENV });
const db = cloud.database();

// 创建集合
async function createCollections() {
  const collections = [
    'users',
    'checkins',
    'museums',
    'themeParks',
    'userFavorites',
    'dataSyncLog'
  ];
  
  for (const collection of collections) {
    try {
      await db.createCollection(collection);
      console.log(`✅ 集合 ${collection} 创建成功`);
    } catch (error) {
      console.error(`❌ 集合 ${collection} 创建失败`, error);
    }
  }
}

// 创建索引
async function createIndexes() {
  const indexes = [
    // users 集合索引
    {
      collection: 'users',
      indexName: 'openid',
      fields: ['openid'],
      unique: true
    },
    // checkins 集合索引
    {
      collection: 'checkins',
      indexName: 'userId',
      fields: ['userId']
    },
    {
      collection: 'checkins',
      indexName: 'userId_visitDate',
      fields: ['userId', 'visitDate']
    },
    {
      collection: 'checkins',
      indexName: 'timeline',
      fields: ['timeline.year', 'timeline.month']
    },
    // museums 集合索引
    {
      collection: 'museums',
      indexName: 'province_city',
      fields: ['province', 'city']
    },
    {
      collection: 'museums',
      indexName: 'level',
      fields: ['info.level']
    },
    {
      collection: 'museums',
      indexName: 'type',
      fields: ['info.type']
    },
    // themeParks 集合索引
    {
      collection: 'themeParks',
      indexName: 'brand',
      fields: ['brand']
    },
    {
      collection: 'themeParks',
      indexName: 'province_city',
      fields: ['location.province', 'location.city']
    },
    // userFavorites 集合索引
    {
      collection: 'userFavorites',
      indexName: 'userId',
      fields: ['userId']
    },
    {
      collection: 'userFavorites',
      indexName: 'userId_placeType_placeId',
      fields: ['userId', 'placeType', 'placeId'],
      unique: true
    },
    // dataSyncLog 集合索引
    {
      collection: 'dataSyncLog',
      indexName: 'dataType',
      fields: ['dataType']
    },
    {
      collection: 'dataSyncLog',
      indexName: 'syncTime',
      fields: ['syncTime']
    }
  ];
  
  for (const index of indexes) {
    try {
      await db.collection(index.collection).createIndex(index.indexName, {
        fields: index.fields,
        unique: index.unique || false
      });
      console.log(`✅ 索引 ${index.collection}.${index.indexName} 创建成功`);
    } catch (error) {
      console.error(`❌ 索引 ${index.collection}.${index.indexName} 创建失败`, error);
    }
  }
}

// 执行初始化
exports.main = async (event, context) => {
  try {
    await createCollections();
    await createIndexes();
    
    return {
      success: true,
      message: '数据库初始化完成'
    };
  } catch (error) {
    return {
      success: false,
      error: error.message
    };
  }
};
```

    const themeParks = require('../../data/theme-parks.json');

    await db.collection('themeParks').where({}).remove();
    
    // 3. 批量写入新数据
    for (let i = 0; i < themeParks.length; i += 100) {
      const batch = themeParks.slice(i, i + 100);
      await db.collection('themeParks').add({ data: batch });
    }
    
    // 4. 记录同步日志
    await db.collection('dataSyncLog').add({
      data: {

        successCount: themeParks.length,
        failedCount: 0,
        syncTime: new Date().toISOString(),
        duration: (Date.now() - startTime) / 1000
      }
    });
    
    return {
      success: true,
      count: themeParks.length,
      duration: (Date.now() - startTime) / 1000
    };
    
  } catch (error) {
    return {
      success: false,
      error: error.message
    };

### 必填字段校验

function validateCheckin(data) {
  const required = [
    'userId',
    'placeType',
    'placeId',
    'placeName',
    'city',
    'province',
    'visitDate',
    'location'
  ];
  
  for (const field of required) {
    if (!data[field]) {
      throw new Error(`必填字段 ${field} 缺失`);
    }
  }
  
  // 日期格式校验
  if (!/^\d{4}-\d{2}-\d{2}$/.test(data.visitDate)) {
    throw new Error('日期格式不正确，应为 YYYY-MM-DD');
  }
  
  // 照片数量校验
  if (data.photos && data.photos.length > 9) {
    throw new Error('照片数量不能超过 9 张');
  }
  
  // 笔记长度校验
  if (data.notes && data.notes.length > 500) {
    throw new Error('笔记长度不能超过 500 字');
  }
  
  return true;
}
