### 5.1 数据验证测试用例

```javascript
// test/validate-theme-parks.test.js

const validateThemeParks = require('../scripts/init-database.js').validateThemeParks;

describe('游乐场数据验证', () => {
  test('TC-DV-01: 空数据应抛出错误', () => {
    expect(() => validateThemeParks([])).toThrow('游乐场数据必须是数组');
  });

  test('TC-DV-02: 缺少必填字段应抛出错误', () => {
    const invalidData = [{ _id: 'test', brand: 'test' }];
    expect(() => validateThemeParks(invalidData)).toThrow('缺少必填字段');
  });

  test('TC-DV-03: 坐标格式错误应抛出错误', () => {
    const invalidData = [{
      _id: 'test',
      brand: 'test',
      name: 'test',
      location: { coordinates: [22.5640, '114.与其他'] },
      facilities: [],
      ticketInfo: {},
      openTime: '09:00-18:00'
    }];
    expect(() => validateThemeParks(invalidData)).toThrow('坐标数据不正确');
  });

  test('TC-DV-04: 合法数据应通过验证', () => {
    const validData = [{
      _id: 'themePark_1',
      brand: '迪士尼',
      name: '上海迪士尼乐园',
      location: { coordinates: [31.1434, 121.6570] },
      facilities: ['主题园区'],
      ticketInfo: { adult: '399元' },
      openTime: '09:00-21:00'
    }];
    expect(validateThemeParks(validData)).toBe(true);
  });
});
```
